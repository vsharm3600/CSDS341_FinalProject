given a user types their student id, return all people who have tested positive who live in the same residence hall
SELECT s.sid
FROM STUDENT s, STUDENT s2, TESTS t
WHERE t.sid = s.sid
AND t.has_covid = 1
AND s2.sid = "input"
AND s2.address = s.address
